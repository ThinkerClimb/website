主程式(mcu8051ide編譯) 

  port3 equ 0b0h
  port5 equ 0c8h
  bit2  equ 4h
  bit3  equ 8h
  bit4  equ 10h
  bit5  equ 20h
  
  .org 0h
  jmp _start

  .org 100h
_start:
  mov port3, #0ffh
  mov port5, #0ffh
key0_chk:
  mov a, port5
  anl a, #bit4
  jz led0_on
  orl port5, #bit5
  jmp key1_chk
led0_on:
  mov a, #bit5
  xrl a, #0ffh
  anl port5, a
key1_chk:
  mov a, port3
  anl a, #bit3
  jz led1_on
  orl port3, #bit2
  jmp final
led1_on:
  mov a, #bit2
  xrl a, #0ffh
  anl port3, a
final:
  jmp _start
  .end

In2 ⇒ Out2
