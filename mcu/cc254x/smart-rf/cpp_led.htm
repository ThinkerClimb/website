/*======================================================================
 LED blink demo for Smart-RF development board
 Steward Fu
 20161009
======================================================================*/
#include "../../../include/cc2540.h"

// define LED pinout
#define LED0 P1_0
#define LED1 P1_1
#define LED2 P1_2
#define LED3 P1_3

/*======================================================================
description: 
 delay in ms level

paramters:
 cnt:
  count

returned:
 nothing
======================================================================*/
void delay(unsigned int cnt)
{ 
 unsigned int ms=0;
 while(cnt--){
  for(ms=0; ms<512; ms++);
 } 
}

/*======================================================================
description:
 set LED state as ON or OFF

paramters:
 mode:
  0: ON
  1: OFF

returned:
 nothing
======================================================================*/
void SetLED(unsigned char mode)
{   
 LED0 = mode;
 LED1 = mode;
 LED2 = mode;
 LED3 = mode;
}

/*======================================================================
description:
 initialize GPIO

parameters:
 none

returned:
 nothing
======================================================================*/
void InitGPIO(void)
{
 P1DIR|= 0x0f;
 P1SEL&= ~0x0f;
 SetLED(0);
}

/*======================================================================
description:
 main entry point

parameters:
 none

returned:
 nothing
======================================================================*/
void main(void)
{
 unsigned char idx=0;

 InitGPIO();     
 while(1){
  idx^= 1;
  SetLED(idx);
  delay(1000);
 }
}

CC=sdcc
all:
  $(CC) main.c
  packihx main.ihx > main.hex
clean:
  rm -rf main.lk
  rm -rf main.map
  rm -rf main.rel
  rm -rf main.sym
  rm -rf main.ihx
  rm -rf main.hex
  rm -rf main.lst
  rm -rf main.mem
  rm -rf main.rst
