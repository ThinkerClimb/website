#include "../../../include/cc2540.h"

// define button pinout
#define BTN0 P0_0
#define BTN1 P0_1
#define BTN2 P0_4
#define BTN3 P0_5

// define led pinout
#define LED0 P1_0
#define LED1 P1_1
#define LED2 P1_2
#define LED3 P1_3

/*======================================================================
description:
 initialize GPIO

parameters:
 none

returned:
 nothing
======================================================================*/
void InitGPIO(void)
{
 P0DIR|= 0x33;
 P0SEL&= ~0x33;
 P1DIR|= 0x0f;
 P1SEL&= ~0x0f;
}

/*======================================================================
description:
 main entry point

parameters:
 none

returned:
 nothing
======================================================================*/
void main(void)
{
 InitGPIO();     
 while(1){
  LED0 = BTN0;
  LED1 = BTN1;
  LED2 = BTN2;
  LED3 = BTN3;
 }
}

CC=sdcc
all:
  $(CC) main.c
  packihx main.ihx > main.hex
clean:
  rm -rf main.lk
  rm -rf main.map
  rm -rf main.rel
  rm -rf main.sym
  rm -rf main.ihx
  rm -rf main.hex
  rm -rf main.lst
  rm -rf main.mem
  rm -rf main.rst
