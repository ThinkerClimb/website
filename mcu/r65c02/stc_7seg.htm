R65C02 >> STC15F204EA

七段顯示器測試

完成硬體電路後，接著就需要驗證每個元件的功能是否正常，才可以確保硬體電路正常執行，為此，目前先測試七段顯示器是否可以正常運作，如下程式碼會在七段顯示器上，顯示0~9數值。

main.s 

  org 0000h
  jmp start

  org 0100h
start:
  mov r0, #0
loop:
  mov a, r0
  call show_7seg
  call delay_1s
  inc r0
  cjne r0, #10, loop
  jmp start

delay_20m: 
  mov r6, #250
  mov r7, #250
d1: 
  djnz r6, d1
  djnz r7, d1
  ret

delay_1s: 
  mov r5, #50
d2: 
  call delay_20m
  djnz r5, d2
  ret

; p2.0 -> b
; p2.1 -> e
; p2.2 -> d
; p2.3 -> c
; p3.5 -> g
; p3.6 -> f
; p3.7 -> a
show_7seg:
  mov dptr, #table
  push acc
  clr c
  rlc a
  movc a, @a+dptr
  mov p2, a
  pop acc
  clr c
  rlc a
  inc a
  movc a, @a+dptr
  mov p3, a
  ret

table:
  dw 0f02fh ; 0
  dw 0f6ffh ; 1
  dw 0f85fh ; 2
  dw 0f25fh ; 3
  dw 0f69fh ; 4
  dw 0f31fh ; 5
  dw 0f11fh ; 6
  dw 0f66fh ; 7
  dw 0f01fh ; 8
  dw 0f61fh ; 9
  end

Line 6 查表數值。
Line 7~12 顯示七段顯示器的數值，每1秒為間隔。
Line 15~21 Delay 20ms。
Line 23~28 Delay 1s。
Line 37~51 由P2以及P3輸出數值。
Line 51~62 查表。

Makefile 

all:
  mcu8051ide --compile main.s
