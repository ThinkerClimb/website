Qt已經被移植到許多平臺上，而且跑起來的效率看似不錯，加上N900也是使用Qt做爲開發程式，因此，司徒只好又來插花一下，介紹一些關於開發Qt程式的相關文章，而談到Linux視窗程式，另一個可以拿來參考比較的程式則是Gtk程式，司徒認爲，Qt程式比較像Windows的MFC程式；而Gtk程式比較像是Win32 API寫法，因此，一樣的視窗寫法，Qt程式會比Gtk程式簡短些，接著一樣從最基本的Hello, world!學起。 

$ sudo apt-get install libqt4-dev
$ cd ~/Downloads
$ vim main.cpp

#include <QApplication>
#include <QWidget>
#include <QLabel>

int main(int argc, char** argv)
{
  QApplication app(argc, argv);
  QWidget *win;
  QLabel *label;

  win = new QWidget;
  label = new QLabel("Hello, world!", win);
  win->show();
  return app.exec();
}


$ vim main.pro

TARGET = main
SOURCES = main.cpp


$ qmake
$ make
