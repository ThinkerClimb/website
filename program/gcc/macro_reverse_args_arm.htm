Macro反相參數(ARM)

在呼叫副程式的規則中，ARM使用R0~R3做為前四個參數，超過則Push到Stack，但是，呼叫cdecl定義的副程式時，順序必須是反向Push到Stack(從最後一個參數開始Push)，因此，如果在Macro中有呼叫到用vararg當參數的副程式(cdecl定義)時，須反相Push到Stack，但是前四個參數依舊放到R0~R3，這種規則其實蠻難搞的，因此，司徒寫了一個參考範例，使用者可以參考看看： 

  .macro myprintf fmt:req, args:vararg
    .section .rodata
    .align
    _fmtsz: .asciz "\fmt"
    .section .text
    .align
    .ifnb \args
      argc = 0
      .irp param, \args
        argc = argc + 1
      .endr
      argrc = argc
      .rept argc
        argrc = argrc - 1
        argc = 0
        .irp param, \args
          .if argrc == argc
            .if argrc == 0
              mov r1, #\param
            .elseif argrc == 1
              mov r2, #\param
            .elseif argrc == 2
              mov r3, #\param
            .else
              mov r0, #\param
              push {r0}
            .endif
          .endif
          argc = argc + 1
        .endr
      .endr
    .endif
    fmtsz: .word _fmtsz
    ldr r0, fmtsz
    bl printf
  .endm

  .text
  .align
  .global main
  .extern printf
  .extern exit

main:
  myprintf "%d %d %d %d %d %d\n", 1, 2, 3, 4, 5, 6
  mov r0, #0
  bl exit
  .end

編譯測試 

$ gcc -g -o test test.s
$ ./test
$ 1 2 3 4 5 6
