基本上，撰寫Linux視窗程式，常見的大概就屬Qt和Gtk兩類，雖然Gtk比較繁碎一些，包裝並不像Qt那麼精緻，不過，在寫Gtk程式時，總是感覺像是在寫API Level的程式一樣，雖然難一點，不過卻很紮實，因此，司徒還是蠻喜愛Gtk程式，而最簡單的Hello, world!程式，一樣是第一個測試的範例程式： 

$ sudo apt-get install build-essential libgtk2.0-dev pkg-config
$ cd ~/Downloads/
$ vim main.c

#include <gtk/gtk.h>

int main(int argc, char** argv)
{
  GtkWidget *win;
  GtkWidget *label;

  gtk_init(&argc, &argv);
  win=gtk_window_new(GTK_WINDOW_TOPLEVEL);

  label = gtk_label_new("Hello, world!");
  gtk_container_add(GTK_CONTAINER(win), label);

  gtk_widget_show_all(win);
  g_signal_connect(win, "destroy", G_CALLBACK(gtk_main_quit), NULL);
  gtk_main();
  return 0;
}


$ gcc -o main main.c `pkg-config --libs --cflags gtk+-2.0`
